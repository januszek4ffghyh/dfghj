    function markServerFetchFailure(label, error) {
        const tNow = nowMs();
        serverFetchBackoffUntil = Math.max(serverFetchBackoffUntil, tNow + 5_000);
        if (tNow - lastServerFetchErrorLogAt >= 10_000) {
            lastServerFetchErrorLogAt = tNow;
            console.error(label, error);
        }
        updateServerConnectionStatus(false, String(error?.message || error || label));

    async function fetchWithTimeout(url, options = {}, timeoutMs = 6_000) {
        if (typeof AbortController !== "function") return fetch(url, options);

        const controller = new AbortController();
        const timer = window.setTimeout(() => controller.abort(), timeoutMs);
        try {
            return await fetch(url, { ...options, signal: controller.signal });
        } finally {
            window.clearTimeout(timer);
        }
    }

            text.innerText = connected ? "connected" : "offline";
            text.style.color = connected ? "#22c55e" : "#ef4444";

        }

            console.error("ClanHub movement test report failed:", e);

            movementTestFlushInFlight = false;

    async function pullStorageVault() {
        try {
            const res = await fetchWithTimeout(SERVER_STATE_URL, {
                method: "GET",
                headers: getServerHeaders({ "Accept": "application/json" }),
                cache: "no-store",

            storageHydrated = true;
            markServerFetchSuccess();
            return true;
        } catch (e) {
            storageHydrated = true;
            markServerFetchFailure("ClanHub storage pull failed", e);
            return false;
        }
    }

    async function pushStorageVaultNow() {
        if (storagePushInFlight) {
            storagePushRequested = true;
            return storagePushPromise;
        }

        storagePushInFlight = true;
        storagePushPromise = (async () => {
            try {
                do {
                    storagePushRequested = false;
                    const res = await fetchWithTimeout(SERVER_STATE_URL, {
                        method: "PUT",
                        headers: getServerHeaders({ "Content-Type": "application/json", "Accept": "application/json" }),
                        body: JSON.stringify({ state: storageVault }),
                    });
                    if (!res.ok) throw new Error(`HTTP ${res.status}`);
                    markServerFetchSuccess();
                } while (storagePushRequested);
            } catch (e) {
                markServerFetchFailure("ClanHub storage push failed", e);
            } finally {
                storagePushInFlight = false;
                storagePushPromise = null;
            }
        })();
        return storagePushPromise;
    }

            storagePushTimer = 0;

    }

    function classifyMobWt(wt) {
        const value = Number(wt);
        if (!Number.isFinite(value)) return { type: "unknown", stable: false };
        if (value >= 0 && value <= 9) return { type: "normal", stable: true };
        if (value >= 10 && value <= 19) return { type: "elite", stable: true };
        if (value >= 20 && value <= 29) return { type: "elite2", stable: false };
        if (value >= 30 && value <= 79) return { type: "elite3", stable: true };
        if (value >= 80 && value <= 89) return { type: "heros", stable: false };
        if (value >= 90 && value <= 99) return { type: "colossus", stable: true };
        if (value >= 100) return { type: "titan", stable: true };
        return { type: "unknown", stable: false };
    }

    const DEFAULT_ZONE_MOB_TYPES = {
        normal: "attack",
        elite: "attack",
        elite2: "avoid",
        elite3: "avoid",
        heros: "avoid",
        colossus: "avoid",
        titan: "avoid",
    };

    function normalizeZoneMobTypes(value) {
        const raw = value && typeof value === "object" && !Array.isArray(value) ? value : {};
        const out = { ...DEFAULT_ZONE_MOB_TYPES };
        for (const key of Object.keys(DEFAULT_ZONE_MOB_TYPES)) {
            const policy = String(raw[key] || "").trim();
            if (policy === "attack" || policy === "avoid" || policy === "ignore") out[key] = policy;
        }
        return out;
    }

    function collectMobSpawnSightings() {
        if (!window.g || !g.npc || !window.map) return [];

        const sightings = [];
        for (const id in g.npc) {
            const n = g.npc[id];
            if (!n || !(n.type === 2 || n.type === 3)) continue;

            const wtInfo = classifyMobWt(n.wt);
            if (wtInfo.type === "unknown") continue;

            sightings.push({
                npcId: String(id),
                map: String(map.name || ""),
                name: String(n.nick || n.name || ""),
                lvl: Number(n.lvl) || 0,
                wt: Number(n.wt) || 0,
                mobType: wtInfo.type,
                stableSpawn: !!wtInfo.stable,
                x: Number(n.x) || 0,
                y: Number(n.y) || 0,
            });
        }

        return sightings;
    }

    function currentWorld() {

    function normalizeExternalCharacter(raw) {
        const character = raw && typeof raw === "object" && !Array.isArray(raw) ? raw : {};
        const id = String(character.id || "").replace(/\D/g, "");
        const world = String(character.world || "").trim().toLowerCase();
        const nick = String(character.nick || "").trim();
        if (!id || !world || !nick) return null;
        return {
            ...character,
            id,
            world,
            nick,
            lvl: Number(character.lvl) || 0,
        };
    }

    function normalizeExternalTimer(raw) {
        const timer = raw && typeof raw === "object" && !Array.isArray(raw) ? raw : {};
        const npc = timer.npc && typeof timer.npc === "object" && !Array.isArray(timer.npc) ? timer.npc : {};
        const name = String(npc.name || timer.name || "").trim();
        const world = String(timer.world || currentWorld()).trim().toLowerCase();
        if (!name || !world) return null;
        const wtInfo = classifyMobWt(npc.wt ?? timer.wt);
        return {
            ...timer,
            world,
            minSpawnTime: String(timer.minSpawnTime || ""),
            maxSpawnTime: String(timer.maxSpawnTime || ""),
            npc: {
                ...npc,
                name,
                lvl: Number(npc.lvl ?? timer.lvl) || 0,
                wt: Number(npc.wt ?? timer.wt) || 0,
            },
            mobType: wtInfo.type === "unknown" ? String(npc.type || timer.mobType || "").toLowerCase() : wtInfo.type,
        };
    }

    function parseLevelRangeValue(value) {
        const raw = String(value || "").trim();
        if (!raw) return null;
        const range = raw.match(/^(\d+)\s*-\s*(\d+)$/);
        if (range) {
            const min = Number(range[1]);
            const max = Number(range[2]);
            return Number.isFinite(min) && Number.isFinite(max) && max >= min ? { min, max } : null;
        }
        const lvl = Number(raw);
        return Number.isFinite(lvl) && lvl > 0 ? { min: lvl, max: lvl } : null;
    }

    function watcherTimerLevel(timer) {
        return Number(timer?.npc?.lvl ?? timer?.lvl) || 0;

    function watcherTimerName(timer) {
        return String(timer?.npc?.name || timer?.name || "").trim();

    function watcherTimerType(timer) {
        const wtInfo = classifyMobWt(timer?.npc?.wt ?? timer?.wt);
        return wtInfo.type === "unknown" ? String(timer?.npc?.type || timer?.mobType || "").trim().toLowerCase() : wtInfo.type;
    }

    function watcherTimerLocation(timer) {

    }

    function watcherTimerKey(timer) {
        return String(timer?.timerKey || timer?.key || watcherTimerName(timer)).toLowerCase();
    }

    function watcherTimerMinMs(timer) {
        return Date.parse(timer?.minSpawnTime || "");
    }

    function watcherTimerMaxMs(timer) {
        return Date.parse(timer?.maxSpawnTime || "");
    }

    function isWatcherTimerDue(timer, now = Date.now()) {
        const minMs = watcherTimerMinMs(timer);
        return Number.isFinite(minMs) && minMs <= now;
    }

    function chooseWatcherTimer(settings, world = currentWorld()) {
        const now = Date.now();
        const type = String(settings?.huntType || "elite2").trim().toLowerCase();
        const selectedMob = String(settings?.huntMob || "").trim().toLowerCase();
        const candidates = lastLootlogTimers
            .map(normalizeExternalTimer)
            .filter(Boolean)
            .filter((timer) => String(timer.world || "").toLowerCase() === world)
            .filter((timer) => {
                const timerType = watcherTimerType(timer);
                if (type && timerType !== type) return false;
                if (selectedMob && watcherTimerName(timer).toLowerCase() !== selectedMob) return false;
                return !!watcherTimerName(timer);
            })
            .map((timer) => {
                const minMs = watcherTimerMinMs(timer);
                const active = isWatcherTimerDue(timer, now);
                const nextTime = Number.isFinite(minMs) ? minMs : Number.MAX_SAFE_INTEGER;
                return { timer, active, nextTime };
            })
            .sort((a, b) => Number(b.active) - Number(a.active) || a.nextTime - b.nextTime);
        return candidates[0]?.timer || null;
    }

    function parseWatcherClientId(clientId) {
        const match = String(clientId || "").match(/^hero_(\d+)_(\d+)$/);
        return match ? { accountId: match[1], heroId: match[2] } : null;
    }

    function findCharacterByHeroId(heroId) {
        const id = String(heroId || "").replace(/\D/g, "");
        return lastAccountCharacters
            .map(normalizeExternalCharacter)
            .filter(Boolean)
            .find((character) => String(character.id || "") === id) || null;
    }

    function relogToCharacter(character, timer, reason = "") {
        const targetHeroId = String(character?.id || "").replace(/\D/g, "");
        const world = String(character?.world || currentWorld()).replace(/[^a-z0-9_-]/gi, "").toLowerCase();
        if (!targetHeroId || !world) return false;
        const key = `${targetHeroId}|${watcherTimerName(timer)}|${timer?.minSpawnTime || ""}|${reason}`;
        const now = nowMs();
        if (lastWatcherRelogKey === key && now - lastWatcherRelogAt < 30_000) return false;
        lastWatcherRelogKey = key;
        lastWatcherRelogAt = now;
        updateStatus(`Watcher relog: ${character.nick || targetHeroId}${reason ? ` ${reason}` : ""}`, "#66d9ff");
        document.cookie = `mchar_id=${targetHeroId}; domain=.margonem.pl; path=/`;
        window.location.href = `https://${world}.margonem.pl`;
        return true;
    }

    function clientIdForCharacter(character) {
        const account = String(getHeroSnapshot()?.account || character?.accountId || character?.account || "").replace(/\D/g, "");
        const heroId = String(character?.id || "").replace(/\D/g, "");
        return account && heroId ? `hero_${account}_${heroId}` : "";
    }

    function clearWatcherState() {
        storageRemove(WATCHER_STATE_LS);
        watcherActiveKey = "";
        watcherActiveMinSpawnTime = "";
        watcherActiveTargetMap = "";
        watcherAttackStarted = false;
        watcherGroupInviteLastAt = {};
        clearWatcherPatrol();
    }

    function chooseFallbackCharacter(returnClientId = "") {
        const returnHeroId = parseWatcherClientId(returnClientId)?.heroId || "";
        if (returnHeroId) {
            const exact = findCharacterByHeroId(returnHeroId);
            if (exact) return exact;
        }
        return lastAccountCharacters
            .map(normalizeExternalCharacter)
            .filter(Boolean)
            .filter((character) => character.world === currentWorld())
            .find((character) => {
                const settings = lastAccountSettings[clientIdForCharacter(character)];
                if (!settings || settings.enabled !== true || settings.running !== true) return false;
                return settings.mode === "zone" || settings.mode === "hunt";
            }) || null;
    }

    function watcherTaskFromSettings(clientId, settings) {
        const identity = parseWatcherClientId(clientId);
        if (!identity || !settings || settings.enabled !== true || settings.running !== true || settings.mode !== "watcher") return null;
        const character = findCharacterByHeroId(identity.heroId);
        if (!character) return null;
        const timer = chooseWatcherTimer(settings, character.world);
        return {
            clientId,
            heroId: identity.heroId,
            character,
            settings,
            timer,
        };
    }

    function allWatcherTasks() {
        return Object.entries(lastAccountSettings || {})
            .map(([clientId, settings]) => watcherTaskFromSettings(clientId, settings))
            .filter(Boolean);
    }

        return getClientId(getHeroSnapshot());

    function isReturnableCurrentMode() {
        const settings = lastAccountSettings[currentClientId()] || lastWatcherSettings;
        return !!settings && settings.enabled === true && settings.running === true && (settings.mode === "zone" || settings.mode === "hunt");
    }

    function findNextDueWatcherTask(tasks, excludeClientId = "", now = Date.now()) {
        return (Array.isArray(tasks) ? tasks : [])
            .filter((task) => task?.timer && String(task.clientId || "") !== String(excludeClientId || ""))
            .map((task) => ({
                task,
                minMs: watcherTimerMinMs(task.timer),
                maxMs: watcherTimerMaxMs(task.timer),
            }))
            .filter((entry) => Number.isFinite(entry.minMs) && entry.minMs <= now)
            .sort((a, b) => {
                const aMax = Number.isFinite(a.maxMs) ? a.maxMs : Number.MAX_SAFE_INTEGER;
                const bMax = Number.isFinite(b.maxMs) ? b.maxMs : Number.MAX_SAFE_INTEGER;
                return a.minMs - b.minMs || aMax - bMax || String(a.task.clientId || "").localeCompare(String(b.task.clientId || ""));
            })[0]?.task || null;
    }

    function watcherFinishNeedsScatter(reason) {
        const text = String(reason || "").toLowerCase();
        return text.includes("killed") || text.includes("visible done");
    }

    function startWatcherGroupScatter(reason, state = readWatcherState()) {
        const now = Date.now();
        if (window.g && g.party && typeof g.party === "object") {
            if (isCurrentHeroPartyLeader()) disbandCurrentWatcherGroup("Watcher group disband");
            else leaveCurrentWatcherGroup("Watcher group leave");
        }

        watcherAttackStarted = false;
        B = undefined;
        A = false;
        z = false;
        window.road = [];
        writeWatcherState({
            ...state,
            cleanupAfterKill: true,
            cleanupReason: String(reason || "killed").slice(0, 120),
            groupCleanedAt: now,
            attackStarted: false,
            updatedAt: new Date().toISOString(),
        });
        refreshResolvedTravelTarget(true);
        moveToWatcherWaitSpot("Watcher scatter");
        return true;
    }

    function handleWatcherScatterState(task, state = readWatcherState(), tasks = allWatcherTasks()) {
        if (!state.cleanupAfterKill) return false;
        const reason = String(state.cleanupReason || "killed");
        const targetName = task?.timer ? watcherTimerName(task.timer) : String(task?.settings?.huntMob || huntMobTarget || "").trim();
        const targetType = task?.timer ? watcherTimerType(task.timer) : String(task?.settings?.huntType || huntTypeTarget || "elite2").trim().toLowerCase();
        routeMode = "watcher";
        huntMobTarget = targetName;
        huntTypeTarget = targetType;
        watcherAttackStarted = false;
        refreshResolvedTravelTarget(false);

        const cleanedAt = Number(state.groupCleanedAt) || 0;
        if (Date.now() - cleanedAt < WATCHER_GROUP_SCATTER_MS) {
            moveToWatcherWaitSpot("Watcher scatter");
            return true;
        }

        return finishWatcherTask(reason, { ...state, cleanupAfterKill: false, scatterDone: true }, tasks);
    }

    function finishWatcherTask(reason, state = readWatcherState(), tasks = allWatcherTasks()) {
        if (watcherFinishNeedsScatter(reason) && !state.scatterDone && !state.cleanupAfterKill) {
            return startWatcherGroupScatter(reason, state);
        }

        const activeClientId = String(state.watcherClientId || "");
        watcherAttackStarted = false;
        clearWatcherState();
        resetServerRouteState();
        clearMovementState();
        huntMobTarget = "";
        huntTypeTarget = "";
        huntSpawnKey = "";

        const nextTask = findNextDueWatcherTask(tasks, activeClientId);
        if (nextTask) {
            updateStatus(`Watcher next: ${watcherTimerName(nextTask.timer)}`, "#66d9ff");
            return runWatcherTask(nextTask, { returnClientId: state.returnClientId || "" }, tasks);
        }

        const fallback = chooseFallbackCharacter(state.returnClientId || "");
        if (fallback && String(getHeroSnapshot()?.id || "") !== String(fallback.id || "")) {
            updateStatus(`Watcher done: ${reason}`, "#66d9ff");
            return relogToCharacter(fallback, { name: reason, minSpawnTime: `fallback:${Date.now()}` }, "back");
        }
        updateStatus(`Watcher done: ${reason}`, "#66d9ff");
        return true;
    }

    function findVisibleWatcherMob(name, type = "elite2") {
        const target = String(name || "").trim().toLowerCase();
        if (!target || !window.g || !g.npc) return null;
        for (const id in g.npc) {
            const n = g.npc[id];
            if (!n || !(n.type === 2 || n.type === 3)) continue;
            const npcName = String(n.nick || n.name || "").trim().toLowerCase();
            if (npcName !== target) continue;
            const wtInfo = classifyMobWt(n.wt);
            if (type && wtInfo.type !== type) continue;
            return { id: String(id), npc: n };
        }
        return null;
    }

    function visibleMobWaitReady(task) {
        const targetName = String(task?.settings?.huntMob || "").trim();
        const targetType = String(task?.settings?.huntType || "elite2").trim().toLowerCase();
        const visible = findVisibleWatcherMob(targetName, targetType);
        if (!visible) {
            const state = readWatcherState();
            if (state.visibleMobName === targetName) {
                delete state.visibleMobName;
                delete state.visibleMobFirstSeenAt;
                writeWatcherState(state);
            }
            return false;
        }

        const state = readWatcherState();
        const now = nowMs();
        if (state.visibleMobName !== targetName || !Number.isFinite(Number(state.visibleMobFirstSeenAt))) {
            writeWatcherState({ ...state, visibleMobName: targetName, visibleMobFirstSeenAt: now });
            updateStatus(`Watcher sees ${targetName}: wait 45s`, "#e7b54a");
            return false;
        }

        const age = now - Number(state.visibleMobFirstSeenAt);
        if (age < 45_000) {
            updateStatus(`Watcher sees ${targetName}: wait ${Math.ceil((45_000 - age) / 1000)}s`, "#e7b54a");
            return false;
        }
        return true;
    }

    function runWatcherTask(task, state = readWatcherState(), tasks = allWatcherTasks()) {
        const timer = task.timer;
        const character = task.character;
        const timerKey = watcherTimerKey(timer);
        const minSpawnTime = String(timer.minSpawnTime || "");
        const attackStarted = state.attackStarted === true || watcherAttackStarted;
        if (state.active && String(state.watcherClientId || "") === String(task.clientId || "") && state.minSpawnTime && minSpawnTime && state.minSpawnTime !== minSpawnTime) {
            return finishWatcherTask(`killed: ${watcherTimerName(timer)}`, state, tasks);
        }

        const now = Date.now();
        const minMs = watcherTimerMinMs(timer);
        const maxMs = watcherTimerMaxMs(timer);
        const attackAt = minMs;
        const targetMap = watcherTimerLocation(timer);

        const heroSnapshot = getHeroSnapshot();
        if (String(heroSnapshot?.id || "") !== String(character.id || "")) {
            writeWatcherState({
                active: true,
                watcherClientId: task.clientId,
                returnClientId: state.returnClientId || (isReturnableCurrentMode() ? currentClientId() : ""),
                timerKey,
                minSpawnTime,
                targetMap,
                attackStarted: false,
                updatedAt: new Date().toISOString(),
            });
            return relogToCharacter(character, timer, "to-watch");
        }

        watcherActiveKey = timerKey;
        watcherActiveMinSpawnTime = minSpawnTime;
        const targetName = watcherTimerName(timer);
        const targetType = watcherTimerType(timer);
        const previousTargetMap = watcherActiveTargetMap;
        watcherActiveTargetMap = targetMap;
        if (attackStarted && Number.isFinite(maxMs) && now >= maxMs + WATCHER_EXPIRED_GRACE_MS && !findVisibleWatcherMob(targetName, targetType)) {
            return finishWatcherTask(`timer expired: ${targetName}`, state, tasks);
        }

        const changed = routeMode !== "watcher" || huntMobTarget !== targetName || huntTypeTarget !== targetType || previousTargetMap !== targetMap;
        routeMode = "watcher";
        huntMobTarget = targetName;
        huntTypeTarget = targetType;
        if (changed) {
            huntSpawnKey = "";
            resetServerRouteState();
            clearMovementState();
        }
        if (handleWatcherScatterState(task, state, tasks)) return true;
        const waitingForAttack = Number.isFinite(attackAt) && now < attackAt;

        writeWatcherState({
            ...state,
            active: true,
            watcherClientId: task.clientId,
            timerKey,
            minSpawnTime,
            targetMap,
            attackStarted: waitingForAttack ? false : watcherAttackStarted || state.attackStarted === true,
            updatedAt: new Date().toISOString(),
        });

        if (waitingForAttack) {
            huntSpawnKey = "";
            watcherAttackStarted = false;
            updateStatus(`Watcher waits min: ${watcherTimerName(timer)}`, "#e7b54a");
            refreshResolvedTravelTarget(true);
            updateRouteInfo();
            return true;
        }

        watcherAttackStarted = true;
        writeWatcherState({ ...readWatcherState(), attackStarted: true, updatedAt: new Date().toISOString() });
        updateStatus(`Watcher hunt: ${targetName}`, "#66d9ff");
        updateRouteInfo();
        return true;
    }

    function applyWatcherAutomation() {
        const state = readWatcherState();
        const tasks = allWatcherTasks();
        const currentHeroId = String(getHeroSnapshot()?.id || "");
        const currentTask = tasks.find((task) => task.heroId === currentHeroId);

        if (state.active && state.watcherClientId) {
            const activeTask = tasks.find((task) => task.clientId === state.watcherClientId);
            if (activeTask && state.cleanupAfterKill) {
                return handleWatcherScatterState(activeTask, state, tasks);
            }
            if (activeTask && String(state.timerKey || "").startsWith("visible:") && state.attackStarted === true) {
                if (handleWatcherScatterState(activeTask, state, tasks)) return true;
                const targetName = String(activeTask.settings.huntMob || "").trim();
                const targetType = String(activeTask.settings.huntType || "elite2").trim().toLowerCase();
                if (activeTask.timer) {
                    const nextMinSpawnTime = String(activeTask.timer.minSpawnTime || "");
                    return finishWatcherTask(
                        nextMinSpawnTime ? `killed: ${watcherTimerName(activeTask.timer)}` : `new timer: ${watcherTimerName(activeTask.timer)}`,
                        state,
                        tasks,
                    );
                }
                if (!findVisibleWatcherMob(targetName, targetType)) {
                    return finishWatcherTask(`visible done: ${targetName}`, state, tasks);
                }
                updateStatus(`Watcher hunt visible: ${targetName}`, "#66d9ff");
                return true;
            }
            if (activeTask?.timer) return runWatcherTask(activeTask, state, tasks);
            if (activeTask) {
                return finishWatcherTask(`timer missing: ${activeTask.settings?.huntMob || "watcher"}`, state, tasks);
            }
            return finishWatcherTask("watcher disabled", state, tasks);
        }

        if (currentTask && !currentTask.timer && visibleMobWaitReady(currentTask)) {
            const targetName = String(currentTask.settings.huntMob || "").trim();
            const targetType = String(currentTask.settings.huntType || "elite2").trim().toLowerCase();
            routeMode = "watcher";
            huntMobTarget = targetName;
            huntTypeTarget = targetType;
            watcherAttackStarted = true;
            writeWatcherState({
                active: true,
                watcherClientId: currentTask.clientId,
                returnClientId: state.returnClientId || (isReturnableCurrentMode() ? currentClientId() : ""),
                timerKey: `visible:${targetName.toLowerCase()}`,
                minSpawnTime: "",
                attackStarted: true,
                updatedAt: new Date().toISOString(),
            });
            updateStatus(`Watcher hunt visible: ${targetName}`, "#66d9ff");
            return true;
        }

        const dueTask = findNextDueWatcherTask(tasks);
        if (dueTask) return runWatcherTask(dueTask, state, tasks);

        if (currentTask?.timer) return runWatcherTask(currentTask, state, tasks);
        return false;
    }

    function applyServerSettings(settings) {
        const next = normalizeServerSettings(settings);
        if (!next) return;
        const rawMode = String(next.mode || "zone");
        const nextMode = rawMode === "goToMap" || rawMode === "hunt" || rawMode === "explorer" || rawMode === "watcher" ? rawMode : "zone";
        const nextGoToMap = String(next.goToMap || "").trim();
        const nextHuntType = String(next.huntType || "").trim();
        const nextHuntMob = String(next.huntMob || "").trim();
        const nextExplorerTarget = String(next.explorerTarget || "").trim();
        const nextZoneMobTypes = normalizeZoneMobTypes(next.zoneMobTypes);
        const hasGoToX = next.goToMapX !== null && next.goToMapX !== undefined && next.goToMapX !== "";
        const hasGoToY = next.goToMapY !== null && next.goToMapY !== undefined && next.goToMapY !== "";
        const nextGoToX = Number(next.goToMapX);
        const nextGoToY = Number(next.goToMapY);
        const normalizedGoToX = hasGoToX && Number.isFinite(nextGoToX) ? Math.trunc(nextGoToX) : null;
        const normalizedGoToY = hasGoToY && Number.isFinite(nextGoToY) ? Math.trunc(nextGoToY) : null;
        const zoneMobTypesChanged = JSON.stringify(zoneMobTypes) !== JSON.stringify(nextZoneMobTypes);
        const routeChanged =
            lastWatcherSettings?.mode !== nextMode ||
            (nextMode === "watcher" && String(lastWatcherSettings?.huntMob || "") !== nextHuntMob) ||
            (nextMode === "watcher" && String(lastWatcherSettings?.huntType || "") !== nextHuntType) ||
            (nextMode === "watcher" && String(lastWatcherSettings?.levelRange || "") !== String(next.levelRange || "").trim()) ||
            (nextMode !== "watcher" && routeMode !== nextMode) ||
            goToMapTarget !== nextGoToMap ||
            (nextMode !== "watcher" && huntTypeTarget !== nextHuntType) ||
            (nextMode !== "watcher" && huntMobTarget !== nextHuntMob) ||
            explorerTarget !== nextExplorerTarget ||
            zoneMobTypesChanged ||
            goToTargetX !== normalizedGoToX ||
            goToTargetY !== normalizedGoToY;

        if (routeChanged) {
            routeMode = nextMode;
            goToMapTarget = nextGoToMap;
            huntTypeTarget = nextHuntType;
            huntMobTarget = nextHuntMob;
            explorerTarget = nextExplorerTarget;
            zoneMobTypes = nextZoneMobTypes;
            huntSpawnKey = "";
            goToTargetX = normalizedGoToX;
            goToTargetY = normalizedGoToY;
            watcherActiveKey = "";
            watcherActiveMinSpawnTime = "";
            watcherActiveTargetMap = "";
            watcherAttackStarted = false;
            watcherGroupInviteLastAt = {};
            resetServerRouteState();
            clearMovementState();
        }

        lastWatcherSettings = {
            ...next,
            mode: nextMode,
            levelRange: String(next.levelRange || "").trim(),
            huntType: nextHuntType,
            huntMob: nextHuntMob,
        };

        if (typeof next.running === "boolean" && RUNNING !== next.running) {
            setRunning(next.running, next.running ? "Started from ClanHub" : "Stopped from ClanHub");
        }

        const levelRange = String(next.levelRange || "").trim();
        if (serverLevelRange !== levelRange) {
            serverLevelRange = levelRange;
            zoneTravelTargetMap = "";
            lastLvlFilter = "__changed__";
            clearMovementState();
            resetSeenMeta();
            seenMobMeta = loadSeenMeta();
            resetPickerState();
        }

        const watcherHandled = applyWatcherAutomation();
        if (routeMode === "goToMap" || routeMode === "hunt" || routeMode === "explorer") {
            updateRouteInfo();
            if (watcherHandled) return;
        }
    }

    function reportBlockedGatewayForCurrentHero(gatewayKey, reason) {
        const heroSnapshot = getHeroSnapshot();
        const clientId = getClientId(heroSnapshot);
        if (!clientId || !gatewayKey) return;

        fetch(SERVER_BLOCKED_GATEWAY_URL, {
            method: "PUT",
            headers: getServerHeaders({ "Content-Type": "application/json", "Accept": "application/json" }),
            body: JSON.stringify({
                clientId,
                key: gatewayKey,
                reason: String(reason || "").slice(0, 180),
            }),
        }).catch((err) => {
            console.error("ClanHub blocked gateway report failed:", err);
        });
    }

    function blockGatewayForCurrentHero(gatewayKey, reason = "Gateway blocked for this hero -> trying another route") {
        if (!gatewayKey) return false;
        blockedGatewayUntil[gatewayKey] = nowMs() + BLOCKED_GATEWAY_MS;
        gatewayFailedAttempts[gatewayKey] = 0;
        updateStatus(reason, "#FFAA00");
        reportBlockedGatewayForCurrentHero(gatewayKey, reason);
        clearPendingGatewayLock();
        clearGatewayStandingWatch();
        resetServerRouteState();
        clearMovementState();
        refreshResolvedTravelTarget(true);
        return true;
    }

    function watchGatewayStanding(gatewayKey) {
        if (!gatewayKey || !window.map || !resolvedGateway || !window.hero) {
            clearGatewayStandingWatch();
            return false;
        }

        const { x: gx, y: gy } = getGatewayUseTarget(resolvedGateway);
        if (!Number.isFinite(gx) || !Number.isFinite(gy) || hero.x !== gx || hero.y !== gy) {
            clearGatewayStandingWatch();
            return false;
        }

        const tNow = nowMs();
        if (gatewayStandingKey !== gatewayKey) {
            gatewayStandingKey = gatewayKey;
            gatewayStandingSince = tNow;
            return false;
        }

        if (tNow - gatewayStandingSince >= GATEWAY_STANDING_BLOCK_MS) {
            return blockGatewayForCurrentHero(gatewayKey, "Standing on blocked gateway -> trying another route");
        }

        return false;
    }

            if (!until || tNow >= until) delete blockedGatewayUntil[key];

    }

    function handlePendingGatewayTimeout() {
        if (!pendingGatewayKey) return false;

        if (pendingGatewayFromMap && window.map?.name && map.name !== pendingGatewayFromMap) {
            gatewayFailedAttempts[pendingGatewayKey] = 0;
            clearPendingGatewayLock();
            return false;
        }

        if (nowMs() < pendingGatewayUntil) return false;

        const key = pendingGatewayKey;
        const requiresKey = pendingGatewayRequiresKey;
        clearPendingGatewayLock();

        gatewayFailedAttempts[key] = (gatewayFailedAttempts[key] || 0) + 1;
        const maxAttempts = requiresKey ? KEY_GATEWAY_MAX_FAILED_ATTEMPTS : GATEWAY_MAX_FAILED_ATTEMPTS;
        if (gatewayFailedAttempts[key] >= maxAttempts) {
            return blockGatewayForCurrentHero(
                key,
                requiresKey ? "Key gateway failed -> trying another route" : "Gateway did not pass -> trying another route",
            );
        }

        updateStatus(requiresKey ? "Key gateway did not pass -> retry once" : "Gateway did not pass -> retrying", "#FFAA00");
        return false;
    }

    function parseGoldValue(value) {
        if (value === null || value === undefined || value === "") return null;
        const numeric = Number(String(value).replace(/[^\d.-]/g, ""));
        return Number.isFinite(numeric) ? Math.max(0, Math.trunc(numeric)) : null;
    }

    function handleServerWorkflowTravel(workflow) {
        if (!workflow?.targetMap || !window.map || !window.hero) return true;

        if (String(map.name || "") === String(workflow.targetMap || "") && canReachWorkflowInteractSpot(workflow)) {
            if (isHeroWorkflowInteractReady(workflow)) {
                rememberWorkflowResult(workflow, { ok: true, reason: "arrived" });
                updateStatus(workflow.status || "Workflow: arrived", "#e7b54a");
                refreshResolvedTravelTarget(true);
                return true;
            }
            B = undefined;
            A = false;
            z = false;
            updateStatus(`${workflow.status || "Workflow travel"} ${workflow.targetX},${workflow.targetY}`, "#00FFFF");
            if (!hasStoredPathFor(Number(workflow.targetX), Number(workflow.targetY))) {
                window.road = [];
                storePath(Number(workflow.targetX), Number(workflow.targetY));
            }
            return true;
        }

        const gw = v.findBestGw();
        if (!gw) {
            clearMovementState();
            updateStatus(`Workflow: routing to ${workflow.targetMap}`, "#FFCC00");
            return true;
        }

        if (isMapSettling()) {
            clearMovementState();
            updateStatus("Workflow: waiting map", "#FFCC00");
            return true;
        }

        const { x: goX, y: goY } = getGatewayUseTarget(gw);
        const gatewayKey = getResolvedGatewayKey();
        updateStatus(`Workflow: ${resolvedGateway?.targetMap || resolvedNextMap || workflow.targetMap}`, "#00FFFF");

        if (watchGatewayStanding(gatewayKey)) return true;

        if (isHeroAtGatewayUseTile(gw)) {
            const tNow = nowMs();
            if (tNow - lastGatewayUseAt >= GATEWAY_USE_COOLDOWN_MS) {
                lastGatewayUseAt = tNow;

                pendingGatewayUntil = tNow + PENDING_GATEWAY_MS;

                window.road = [];
                if (typeof window._g === "function") _g("walk");
            }
            return true;
        }

        if (!hasPendingGatewayLock() || pendingGatewayKey !== gatewayKey) {
            if (!hasStoredPathFor(goX, goY)) {
                storeGatewayPathOrTryAlternative(gatewayKey, goX, goY, "Workflow");
            }
        }

        return true;
    }

    function shouldCloseShopAfterGoldCheck(previous, current) {
        if (!previous || !current) return false;
        const previousPhase = String(previous.phase || "");
        const currentAction = String(current.action || "");
        if (!["sellWait", "preSellWait", "lootSellWait"].includes(previousPhase)) return false;
        if (currentAction === "wait") return false;
        return current.finished === true || currentAction === "travel" || currentAction === "teleport" || currentAction === "finish";
    }

    function handleServerWorkflow() {
        const workflow = activeServerWorkflow;
        if (!workflow?.active) {
            lastServerWorkflowDirective = null;
            return false;
        }
        if (shouldCloseShopAfterGoldCheck(lastServerWorkflowDirective, workflow)) {
            closeShopDialogAfterWorkflow();
        }
        lastServerWorkflowDirective = {
            action: String(workflow.action || ""),
            phase: String(workflow.phase || ""),
            key: String(workflow.key || ""),
            finished: workflow.finished === true,
        };
        B = undefined;
        A = false;
        z = false;

        if (workflow.finished || workflow.action === "finish") return finishServerWorkflowLocally(workflow);
        if (workflow.action === "wait") {
            updateStatus(workflow.status || "Workflow: waiting", "#e7b54a");
            return true;
        }
        if (workflow.action === "travel") return handleServerWorkflowTravel(workflow);
        if (workflow.action === "talk") return executeWorkflowTalk(workflow);
        if (workflow.action === "teleport") return executeWorkflowTeleport(workflow);
        if (workflow.action === "quickSell") return executeWorkflowQuickSell(workflow);
        if (workflow.action === "buyHeal") return executeWorkflowBuyHeal(workflow);
        if (workflow.action === "buyTeleport") return executeWorkflowBuyTeleport(workflow);

        updateStatus(workflow.status || "Workflow: waiting", "#e7b54a");
        return true;
    }

    function getLevelRange() {
        const raw = serverLevelRange.trim();
        if (!raw) return null;

        const rangeMatch = raw.match(/^(\d+)\s*-\s*(\d+)$/);
        if (rangeMatch) {
            const min = parseInt(rangeMatch[1], 10);
            const max = parseInt(rangeMatch[2], 10);
            if (Number.isFinite(min) && Number.isFinite(max) && min > 0 && max >= min) {
                return { min, max, raw: min === max ? String(min) : `${min}-${max}` };
            }
            return null;
        }

        if (!/^\d+$/.test(raw)) return null;
        const n = parseInt(raw, 10);
        if (Number.isFinite(n) && n > 0) return { min: n, max: n, raw: String(n) };
        return null;
    }

    function addStatusEvent(message) {
        const text = String(message || "").trim();
        if (!text) return;

        const key = text.toLowerCase();
        const tNow = nowMs();
        const busyLike =
            key.includes("walking") ||
            key.includes("traveling") ||
            key.includes("target") ||
            key.includes("pathing") ||
            key.includes("going");
        const minGapMs = busyLike ? 8000 : 1800;
        const lastAt = Number(statusEventLastSeenAt[key] || 0);
        if (tNow - lastAt < minGapMs) return;
        statusEventLastSeenAt[key] = tNow;

        statusEvents.unshift(`${new Date().toISOString().replace("T", " ").replace(/\.\d+Z$/, "Z")} | ${text}`);
        if (statusEvents.length > STATUS_EVENTS_MAX) {
            statusEvents = statusEvents.slice(0, STATUS_EVENTS_MAX);
        }
    }

    function parseGatewayCoord(value) {
        if (typeof value !== "string") return null;
        const parts = value.split(".");
        const x = parseInt(parts[0], 10);
        const y = parseInt(parts[1], 10);
        if (!Number.isFinite(x) || !Number.isFinite(y)) return null;
        return { x, y };
    }

    function findGatewayStandTile(x, y) {
        if (inBounds2(x, y) && isWalkable(x, y)) {
            return { x, y, region: getRegionAt(x, y) };
        }

        const dirs = [
            { dx: -1, dy: 0 },
            { dx: 1, dy: 0 },
            { dx: 0, dy: -1 },
            { dx: 0, dy: 1 },
        ];

        for (let i = 0; i < dirs.length; i++) {
            const nx = x + dirs[i].dx;
            const ny = y + dirs[i].dy;
            if (!inBounds2(nx, ny)) continue;
            if (!isWalkable(nx, ny)) continue;
            return { x: nx, y: ny, region: getRegionAt(nx, ny) };
        }

        return { x: null, y: null, region: null };
    }

    function scanCurrentMapToLocalDb() {
        if (!window.map || !window.g || !map.name || !map.col) return false;

        const db = readLocalMapsDb();
        const gateways = {};
        const gwIds = g.gwIds || {};
        const townname = g.townname || {};

        for (const id in gwIds) {
            const coord = parseGatewayCoord(gwIds[id]);
            if (!coord) continue;
            if (coord.x < 0 || coord.y < 0 || coord.x >= map.x || coord.y >= map.y) continue;

            const targetMap = String(townname[id] || "").trim();
            if (!targetMap) continue;

            const stand = findGatewayStandTile(coord.x, coord.y);
            const tip = getGatewayTip(id);
            gateways[String(id)] = {
                id: String(id),
                x: coord.x,
                y: coord.y,
                targetMap,
                standX: stand.x,
                standY: stand.y,
                region: stand.region,
                requiresKey: gatewayTipRequiresKey(tip),
                tip,
            };
        }

        const nextMapRecord = {
            map: map.name,
            width: Number(map.x) || 0,
            height: Number(map.y) || 0,
            col: String(map.col || ""),
            gateways,
            scannedAt: new Date().toISOString(),
        };

        const previous = db[map.name] || {};
        const changed = JSON.stringify({
            width: previous.width,
            height: previous.height,
            col: previous.col,
            gateways: previous.gateways || {},
        }) !== JSON.stringify({
            width: nextMapRecord.width,
            height: nextMapRecord.height,
            col: nextMapRecord.col,
            gateways: nextMapRecord.gateways,
        });

        db[map.name] = nextMapRecord;
        saveLocalMapsDb(db);
        return changed;
    }

    async function refreshResolvedTravelTarget(force = false) {
        if (nextStepRefreshInFlight) {
            routeFetchPending = true;

            return;

            routeFetchPending = false;
            updateRouteInfo();
            return;
        }

        const needsZone = routeMode !== "goToMap" && routeMode !== "hunt" && routeMode !== "explorer" && routeMode !== "watcher";
        const zoneLevelRange = serverLevelRange.trim();
        const workflowFacts = window.map && window.hero ? getWorkflowInventorySnapshot() : {};

            && (workflowFacts.hasTeleportScroll || String(window.map?.name || "") === SELL_ENTRY_MAP || String(window.map?.name || "") === SELL_TARGET_MAP);
        const workflowNeedsServer = workflowTestOverride?.active
            || activeServerWorkflow?.active
            || workflowFacts.freeSlots === 0
            || Number(workflowFacts.hp) === 1
            || (workflowFacts.healUses !== null && workflowFacts.healUses !== undefined && Number(workflowFacts.healUses) < HEAL_MIN_USES)
            || teleportRestockPossible;
        if (((needsZone && !zoneLevelRange && !workflowNeedsServer) || (routeMode === "goToMap" && !goToMapTarget) || ((routeMode === "hunt" || routeMode === "watcher") && !huntMobTarget)) || !window.map || !window.hero) {
            resetServerRouteState();
            updateRouteInfo();
            return;
        }

        if (!force && needsZone && isMapSettling()) {
            resetServerRouteState();
            updateRouteInfo();
            return;
        }

        if (!force && hasPendingGatewayLock()) {
            updateRouteInfo();
            return;
        }

        const currentMap = map.name;
        const routeKey = routeMode === "goToMap"
            ? `${routeMode}|${goToMapTarget}|${goToTargetX ?? ""}|${goToTargetY ?? ""}`
            : routeMode === "hunt"
                ? `${routeMode}|${huntTypeTarget}|${huntMobTarget}`
                : routeMode === "watcher"
                    ? `${routeMode}|${huntTypeTarget}|${huntMobTarget}`
                    : routeMode === "explorer"
                        ? `${routeMode}|${explorerTarget}`
                        : `${routeMode}|${zoneLevelRange}|${JSON.stringify(zoneMobTypes)}`;
        const workflowKey = [
            workflowTestOverrideKey(),
            activeServerWorkflow?.key || "",
            lastWorkflowActionResult?.key || "",
            workflowFacts.freeSlots ?? "",
            workflowFacts.hp ?? "",
            workflowFacts.gold ?? "",
            workflowFacts.healUses ?? "",
            workflowFacts.teleportUses ?? "",
        ].join("|");
        const key = `${routeKey}|${workflowKey}|${currentMap}|${hero.x}|${hero.y}`;

        if (!force && lastNextStepKey === key && (tNow - lastNextStepFetchAt) < NEXT_STEP_REFRESH_MS) {
            updateRouteInfo();
            return;
        }

        lastNextStepKey = key;
        lastNextStepFetchAt = tNow;
        nextStepRefreshInFlight = true;
        routeFetchPending = true;

        try {
            scanCurrentMapToLocalDb();
            await pushStorageVaultNow();
            let data = null;
            try {
                data = await fetchServerNextStep();
            } catch (serverErr) {
                markServerFetchFailure("ClanHub next-step failed", serverErr);
            }

            if (data?.ok) {
                activeServerWorkflow = data.workflow && data.workflow.active ? data.workflow : null;
                if (workflowTestOverride?.active && activeServerWorkflow?.active) {
                    workflowTestOverride = null;
                }
                lastWorkflowActionResult = null;
                resolvedMode = data.mode || "";
                resolvedPath = Array.isArray(data.path) ? data.path.slice() : [];
                resolvedTarget = data.targetMap || "";
                if (routeMode === "zone") zoneTravelTargetMap = resolvedTarget || zoneTravelTargetMap;
                resolvedNextMap = data.nextMap || "";
                resolvedGateway = data.gateway || null;
                resolvedInsideZone = !!data.insideZone;
                huntTargetX = data.targetX === null || data.targetX === undefined ? null : Number(data.targetX);
                huntTargetY = data.targetY === null || data.targetY === undefined ? null : Number(data.targetY);
                huntSpawnKey = String(data.spawnKey || "");
                routeFetchPending = false;
            } else {
                activeServerWorkflow = null;
                resetServerRouteState();
                routeFetchPending = false;
                if (data) {
                    resolvedMode = data.mode || "";
                    resolvedPath = Array.isArray(data.path) ? data.path.slice() : [];
                    resolvedTarget = data.targetMap || "";
                    resolvedNextMap = data.nextMap || "";
                    resolvedInsideZone = !!data.insideZone;
                }
            }
        } catch (e) {
            console.error("Zone next-step resolution failed:", e);
            activeServerWorkflow = null;
            resetServerRouteState();
            routeFetchPending = false;
        } finally {
            nextStepRefreshInFlight = false;
        }

        updateRouteInfo();
    }

    function findActiveHuntMob() {
        if ((!isHuntMode() && !isWatcherRouteMode()) || !window.g || !g.npc) return null;
        const target = huntMobTarget.trim().toLowerCase();
        if (!target) return null;

        for (const id in g.npc) {
            const n = g.npc[id];
            if (!n || !(n.type === 2 || n.type === 3)) continue;
            const name = String(n.nick || n.name || "").trim().toLowerCase();
            if (name !== target) continue;
            if (huntTypeTarget) {
                const wtInfo = classifyMobWt(n.wt);
                if (wtInfo.type !== huntTypeTarget) continue;
            }
            return String(id);
        }

        return null;
    }

    function watcherGroupEligiblePlayers() {
        if (!window.g || !g.other || typeof g.other !== "object") return [];
        const selfId = currentHeroPartyId();
        const players = [];
        const policyKey = (value) => String(value || "").trim().toLowerCase();
        for (const raw of Object.values(g.other)) {
            if (!raw || typeof raw !== "object") continue;
            const id = String(raw.id || "").replace(/\D/g, "");
            if (!id || id === selfId) continue;
            const relation = Number(raw.relation);
            const playerClanId = String(raw.clan?.id || "").replace(/\D/g, "");
            const playerNick = String(raw.nick || "");
            const playerClanName = String(raw.clan?.name || "");
            const noClan = !playerClanId && !playerClanName.trim();
            const policy = noClan ? String(playerPolicies[id] || playerPolicies[policyKey(playerNick)] || "") : "";
            const clanPolicy = String(playerClanPolicies[policyKey(playerClanName)] || (playerClanId ? playerClanPolicies[playerClanId] : "") || "");
            if (policy === "ignore" || clanPolicy === "ignore") continue;
            const friend = relation === 2;
            const clanMember = clanPolicy === "add";
            if (policy !== "add" && clanPolicy !== "add" && !friend) continue;
            players.push({
                id,
                nick: String(raw.nick || id),
                x: Math.trunc(Number(raw.x) || 0),
                y: Math.trunc(Number(raw.y) || 0),
                relation,
                friend,
                clanMember,
            });
        }
        return players;
    }

        return true;
    }

    function ensureOnlyWatcherGroupMembers(eligiblePlayers = watcherGroupEligiblePlayers()) {

        return false;

    function chooseWatcherPatrolIdleTrigger() {
        return WATCHER_PATROL_IDLE_TRIGGERS[Math.floor(Math.random() * WATCHER_PATROL_IDLE_TRIGGERS.length)] || 100;
    }

    function isWatcherPatrolTile(tile, region) {
        if (!tile || !inBounds2(tile.x, tile.y) || !isWalkable(tile.x, tile.y)) return false;
        if (getRegionAt(tile.x, tile.y) !== region) return false;
        if (isNpcBlockingTile(tile.x, tile.y)) return false;
        if (isGatewayTile(tile.x, tile.y)) return false;
        return true;
    }

    function isGatewayTile(x, y) {
        if (!window.g || !g.gwIds) return false;
        for (const raw of Object.values(g.gwIds)) {
            const coord = parseGatewayCoord(raw);
            if (coord && coord.x === x && coord.y === y) return true;
        }
        return false;
    }

    function chooseWatcherPatrolTarget() {
        if (!window.hero || !window.map) return null;
        const anchor = { x: Number(hero.x) || 0, y: Number(hero.y) || 0 };
        const region = getRegionAt(anchor.x, anchor.y);
        if (region < 0) return null;

        const candidates = [];
        const radius = 5;
        for (let dy = -radius; dy <= radius; dy++) {
            for (let dx = -radius; dx <= radius; dx++) {
                if (!dx && !dy) continue;
                const tile = { x: anchor.x + dx, y: anchor.y + dy };
                if (isWatcherPatrolTile(tile, region)) candidates.push(tile);
            }
        }
        if (!candidates.length) return null;

        for (let i = candidates.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [candidates[i], candidates[j]] = [candidates[j], candidates[i]];
        }

        return candidates[0];
    }

    function handleWatcherPatrol() {
        if (!window.hero || !window.map || B || z) return false;
        const idleCounter = Number(window.g?.idleRequestCounter) || 0;

        if (watcherPatrolTarget && Number(hero.x) === watcherPatrolTarget.x && Number(hero.y) === watcherPatrolTarget.y) {
            clearWatcherPatrol();
            updateStatus(`Watcher wait ${huntMobTarget}: ${map.name} ${hero.x},${hero.y}`, "#e7b54a");
            return true;
        }

        if (!watcherPatrolIdleArmed && idleCounter <= WATCHER_PATROL_IDLE_RESET_TRIGGER) {
            watcherPatrolIdleArmed = true;
            clearWatcherPatrol();
            watcherPatrolIdleTrigger = chooseWatcherPatrolIdleTrigger();
            updateStatus(`Watcher patrol waits idle ${idleCounter}/${watcherPatrolIdleTrigger}`, "#e7b54a");
            return true;
        }

        if (watcherPatrolIdleArmed && idleCounter <= watcherPatrolIdleTrigger) {
            clearWatcherPatrol();
            updateStatus(`Watcher patrol waits idle ${idleCounter}/${watcherPatrolIdleTrigger}`, "#e7b54a");
            return true;
        }

        if (!watcherPatrolTarget) {
            if (!watcherPatrolIdleArmed) {
                updateStatus(`Watcher patrol waits idle reset ${idleCounter}`, "#e7b54a");
                return true;
            }
            watcherPatrolTarget = chooseWatcherPatrolTarget();
            watcherPatrolIdleArmed = false;
        }

        if (!watcherPatrolTarget) {
            updateStatus(`Watcher patrol blocked idle ${idleCounter}`, "#e7b54a");
            return true;
        }

        if (!hasStoredPathFor(watcherPatrolTarget.x, watcherPatrolTarget.y)) {
            B = undefined;
            A = false;
            z = false;
            window.road = [];
            storePath(watcherPatrolTarget.x, watcherPatrolTarget.y);
            if (!window.road || !window.road.length) {
                const failedTarget = watcherPatrolTarget;
                clearWatcherPatrol();
                updateStatus(`Watcher patrol no path: ${failedTarget.x},${failedTarget.y}`, "#e7b54a");
                return true;
            }
        }

        updateStatus(`Watcher patrol idle ${idleCounter}: ${watcherPatrolTarget.x},${watcherPatrolTarget.y}`, "#e7b54a");
        return true;
    }

    function ensureWatcherGroupReadyBeforeAttack() {
        const players = watcherGroupEligiblePlayers();
        if (!ensureOnlyWatcherGroupMembers(players)) return false;
        if (!players.length) return true;

        if (!isCurrentHeroPartyLeader()) {
            updateStatus("Watcher group no leader: patrol", "#e7b54a");
            return true;
        }

        let pending = false;
        for (const player of players) {
            if (isPartyMember(player.id)) continue;

            pending = true;
            const distance = Math.abs(Number(hero.x) - player.x) + Math.abs(Number(hero.y) - player.y);
            if (player.friend || distance <= 1) {
                inviteWatcherGroupPlayer(player);
                updateStatus(`Watcher group invite: ${player.nick}`, "#e7b54a");
                continue;
            }

            B = undefined;
            A = false;
            z = false;
            updateStatus(`Watcher group go to: ${player.nick}`, "#00FFFF");
            if (!hasStoredPathFor(player.x, player.y)) {
                window.road = [];
                storePath(player.x, player.y);
            }
            return false;
        }

        if (pending) {
            B = undefined;
            A = false;
            z = false;
            window.road = [];
            updateStatus("Watcher group: waiting invites", "#e7b54a");
            return false;
        }

        return true;
    }

        if (!isHuntMode() || !window.map || !window.hero) return false;

            Math.abs(Number(hero.x) - huntTargetX) + Math.abs(Number(hero.y) - huntTargetY) <= 1 &&

            window.road = [];
            updateStatus(`Watcher found: ${huntMobTarget}`, "#ff66cc");

        A = false;
        z = false;
        window.road = [];
        updateStatus(`Watcher wait ${huntMobTarget}: ${map.name} ${hero.x},${hero.y}`, "#e7b54a");
        return true;
    }

    function syncDatabase() {
        if (typeof g === "undefined" || !g.npc || !window.map || !window.hero) return;

        const t = nowMs();
        const mapChanged = map.name !== lastMapName;
        if (!mapChanged && t - lastDbSyncAt < DB_SYNC_THROTTLE_MS) return;
        lastDbSyncAt = t;

        if (mapChanged) {
            clearPendingGatewayLock();
            persistentDB = {};
            lastMapName = map.name;
            lastMapChangedAt = t;
            lastLevelZoneClearKey = "";
            resetSeenMeta();
            seenMobMeta = loadSeenMeta();
            resetPickerState();
            resetRegionState();
            resetServerRouteState();
            clearMovementState();
        }

        reportMobSpawns(t);
        reportSeenPlayers(t);

        const range = getLevelRange();
        if (!range) {
            const dbCounter = document.querySelector("#xD_stat_db");
            if (dbCounter) dbCounter.innerText = "0";
            return;
        }

        if (range.raw !== lastLvlFilter) {
            lastLvlFilter = range.raw;
            persistentDB = {};
            clearMovementState();
            resetSeenMeta();
            seenMobMeta = loadSeenMeta();
            resetPickerState();
        }

        const minL = range.min, maxL = range.max;
        if (!seenMobMeta || typeof seenMobMeta !== "object") seenMobMeta = loadSeenMeta();

        for (let id in g.npc) {
            const n = g.npc[id];
            if (
                n &&
                (n.type === 2 || n.type === 3) &&
                zoneMobTypes[classifyMobWt(n.wt).type] === "attack" &&
                n.lvl >= minL &&
                n.lvl <= maxL
            ) {
                persistentDB[id] = { id, x: n.x, y: n.y, lvl: n.lvl, name: n.nick || n.name };

                const meta = seenMobMeta[id];
                if (!meta) seenMobMeta[id] = { firstSeen: t, lastSeen: t };
                else meta.lastSeen = t;
            }
        }

        for (let id in persistentDB) {
            const m = persistentDB[id];
            if (!m || m.lvl < minL || m.lvl > maxL) {
                delete persistentDB[id];
                delete I[id];
                delete unreachableCount[id];
                delete failurePenalty[id];
            }
        }

        for (let id in persistentDB) {
            const m = persistentDB[id];
            const dist = Math.abs(hero.x - m.x) + Math.abs(hero.y - m.y);
            if (dist <= 10 && !g.npc[id]) {
                delete persistentDB[id];
                delete I[id];
                delete unreachableCount[id];
                delete failurePenalty[id];
                if (B == id) {
                    B = undefined;
                    updateStatus("Target vanished, re-scanning", "yellow");
                }
            }
        }

        saveSeenMetaThrottled(seenMobMeta, t);

        const dbCounter = document.querySelector("#xD_stat_db");
        if (dbCounter) dbCounter.innerText = Object.keys(persistentDB).length;
    }

    function calculateDistance(K, L) {
        return Math.abs(K.x - L.x) + Math.abs(K.y - L.y);
    }

    function findPathToLocation(K, L) {
        if (!window.map || !map.col || typeof Pathfinder !== "function") return undefined;

        const start = { x: hero.x, y: hero.y };
        const tryPath = (tx, ty) =>
            new Pathfinder(map.col, map.x, map.y, start, { x: tx, y: ty }, g.npccol).findPath();

        const candidates = [
            { x: K, y: L },
            { x: K + 1, y: L },
            { x: K - 1, y: L },
            { x: K, y: L + 1 },
            { x: K, y: L - 1 },
        ];

        let best = null;
        for (const c of candidates) {
            if (c.x < 0 || c.y < 0 || c.x >= map.x || c.y >= map.y) continue;
            const path = tryPath(c.x, c.y);
            if (path && path.length) {
                if (!best || path.length < best.length) best = path;
            }
        }

        return best || undefined;
    }

    function clearStoredPathRequest() {
        lastStoredPathMap = "";
        lastStoredPathRequestX = null;
        lastStoredPathRequestY = null;
        lastStoredPathRegion = -1;
    }

    function hasStoredPathFor(K, L) {
        return !!(
            window.road &&
            window.road.length &&
            lastStoredPathMap === String(map?.name || "") &&
            lastStoredPathRequestX === K &&
            lastStoredPathRequestY === L &&
            lastStoredPathRegion === getHeroRegion()
        );
    }

    function isNpcBlockingTile(K, L) {
        if (!window.g || !g.npc || !window.map) return false;
        for (let id in g.npc) {
            const n = g.npc[id];
            if (!n) continue;
            if (!shouldTreatNormalMobsAsCollision() && n.wt === 1) continue;
            if (n.type === 4) continue;
            if (n.wt === -2) continue;
            if (Number(n.x) === K && Number(n.y) === L) return true;
        }
        return false;
    }

    function isHighLevelMobForZoneAvoidance(npc) {
        if (!npc || (npc.type !== 2 && npc.type !== 3)) return false;
        const range = getLevelRange();
        if (!range || !Number.isFinite(Number(range.max))) return false;
        return Number(npc.lvl) > Number(range.max);
    }

    function buildHighLevelMobAvoidSet(width, height, start) {
        const set = new Set();
        if (!window.g || !g.npc || !window.map) return set;
        const startX = Number(start?.x);
        const startY = Number(start?.y);

        for (let id in g.npc) {
            const n = g.npc[id];
            if (!isHighLevelMobForZoneAvoidance(n)) continue;

            const mobX = Number(n.x);
            const mobY = Number(n.y);
            if (!Number.isFinite(mobX) || !Number.isFinite(mobY)) continue;

            const startDist = Number.isFinite(startX) && Number.isFinite(startY)
                ? Math.abs(startX - mobX) + Math.abs(startY - mobY)
                : Infinity;

            for (let dx = -HIGH_LEVEL_MOB_AVOID_RADIUS; dx <= HIGH_LEVEL_MOB_AVOID_RADIUS; dx++) {
                for (let dy = -HIGH_LEVEL_MOB_AVOID_RADIUS; dy <= HIGH_LEVEL_MOB_AVOID_RADIUS; dy++) {
                    const dist = Math.abs(dx) + Math.abs(dy);
                    if (dist > HIGH_LEVEL_MOB_AVOID_RADIUS) continue;

                    const x = mobX + dx;
                    const y = mobY + dy;
                    if (x < 0 || y < 0 || x >= width || y >= height) continue;
                    if (x === startX && y === startY) continue;

                    const tileDistFromMob = Math.abs(x - mobX) + Math.abs(y - mobY);
                    if (startDist <= HIGH_LEVEL_MOB_AVOID_RADIUS && tileDistFromMob > startDist) continue;

                    const idx = x + y * width;
                    if (map.col.charAt(idx) === "1") continue;
                    set.add(idx);
                }
            }
        }

        return set;
    }

    function clearRoadIfHighLevelMobTooClose() {
        if (!window.road || !window.road.length || !window.map || !window.hero) return false;
        const avoided = buildHighLevelMobAvoidSet(map.x, map.y, { x: hero.x, y: hero.y });
        if (!avoided.size) return false;

        const touchesDanger = window.road.some((step) => {
            const x = Number(step?.x);
            const y = Number(step?.y);
            if (!Number.isFinite(x) || !Number.isFinite(y)) return false;
            return avoided.has(x + y * map.x);
        });
        if (!touchesDanger) return false;

        window.road = [];
        clearStoredPathRequest();
        updateStatus(`Avoiding high-level mob (${HIGH_LEVEL_MOB_AVOID_RADIUS} tiles)`, "#FFAA00");
        return true;
    }

    function storePath(K, L) {
        const M = findPathToLocation(K, L);
        if (Array.isArray(M)) {
            window.road = M;
            lastStoredPathMap = String(map?.name || "");
            lastStoredPathRequestX = K;
            lastStoredPathRequestY = L;
            lastStoredPathRegion = getHeroRegion();
            movementTestEvent("path", "set road", {
                targetX: K,
                targetY: L,
            });
            return true;
        } else {
            clearStoredPathRequest();
            movementTestEvent("path", "no path", {
                targetX: K,
                targetY: L,
            });
            return false;
        }
    }

    function storeGatewayPathOrTryAlternative(gatewayKey, K, L, reasonPrefix = "Gateway") {
        window.road = [];
        if (storePath(K, L)) return true;
        if (!gatewayKey || !window.map || !window.hero) return false;

        const avoided = buildHighLevelMobAvoidSet(map.x, map.y, { x: hero.x, y: hero.y });
        if (!avoided.size) return false;

        return blockGatewayForCurrentHero(
            gatewayKey,
            `${reasonPrefix}: high-level mob blocks path -> trying another route`,
        );
    }

    class Pathfinder {
        constructor(K, L, M, N, O, P) {
            this.width = L;
            this.height = M;
            this.collisions = this.parseCollisions(K, L, M);
            this.additionalCollisions = P || {};
            this.npcBlocked = this.buildNpcBlockedSet();
            this.highLevelMobAvoided = buildHighLevelMobAvoidSet(L, M, N);

            this.start = this.collisions[N.x][N.y];
            this.end = this.collisions[O.x][O.y];
            this.start.g = 0;
            this.openSet = [this.start];
            this.closedSet = [];
        }

        buildNpcBlockedSet() {
            const set = new Set();
            if (!window.g || !g.npc || !window.map) return set;

            for (let id in g.npc) {
                const n = g.npc[id];
                if (!n) continue;
                if (!shouldTreatNormalMobsAsCollision() && n.wt === 1) continue;
                if (n.type === 4) continue;
                if (n.wt === -2) continue;

                const x = n.x, y = n.y;
                if (x < 0 || y < 0 || x >= map.x || y >= map.y) continue;

                const idx = x + y * map.x;
                if (map.col.charAt(idx) === "1") continue;

                set.add(idx);
            }
            return set;
        }

        parseCollisions(K, L, M) {
            const N = Array(L);
            for (let O = 0; O < L; O++) {
                N[O] = Array(M);
                for (let P = 0; P < M; P++) {
                    N[O][P] = new PathNode(O, P, K.charAt(O + P * L) === "1");
                }
            }
            return N;
        }

        findPath() {
            const TURN_PENALTY = 28;
            const neighbors = [
                { dx: 0, dy: 1, dir: 0 },
                { dx: 0, dy: -1, dir: 1 },
                { dx: 1, dy: 0, dir: 2 },
                { dx: -1, dy: 0, dir: 3 },
            ];

            this.start.g = 0;
            this.start.dir = -1;
            this.start.turns = 0;
            this.start.h = calculateDistance(this.start, this.end);
            this.start.updateGroupMembers = this.start.h;

            while (this.openSet.length > 0) {
                let best = 0;
                for (let i = 1; i < this.openSet.length; i++) {
                    const candidate = this.openSet[i];
                    const currentBest = this.openSet[best];
                    if (
                        candidate.updateGroupMembers < currentBest.updateGroupMembers ||
                        (
                            candidate.updateGroupMembers === currentBest.updateGroupMembers &&
                            candidate.turns < currentBest.turns
                        ) ||
                        (
                            candidate.updateGroupMembers === currentBest.updateGroupMembers &&
                            candidate.turns === currentBest.turns &&
                            candidate.h < currentBest.h
                        )
                    ) {
                        best = i;
                    }
                }

                const cur = this.openSet[best];
                if (cur === this.end) return this.reconstructPath();

                this.openSet.splice(best, 1);
                this.closedSet.push(cur);

                for (let nb of neighbors) {
                    const nx = cur.x + nb.dx, ny = cur.y + nb.dy;
                    if (nx < 0 || nx >= this.width || ny < 0 || ny >= this.height) continue;

                    const node = this.collisions[nx][ny];
                    if (node.collision) continue;

                    const idx = nx + ny * this.width;
                    if (this.npcBlocked.has(idx)) continue;
                    if (this.highLevelMobAvoided.has(idx)) continue;

                    if (this.closedSet.includes(node)) continue;

                    const isTurn = cur.dir !== -1 && cur.dir !== nb.dir;
                    const stepCost = 1 + (isTurn ? TURN_PENALTY : 0);
                    const tentativeG = cur.g + stepCost;
                    const tentativeTurns = cur.turns + (isTurn ? 1 : 0);

                    const inOpen = this.openSet.includes(node);
                    if (!inOpen) this.openSet.push(node);
                    else if (tentativeG > node.g || (tentativeG === node.g && tentativeTurns >= node.turns)) continue;

                    node.previous = cur;
                    node.g = tentativeG;
                    node.dir = nb.dir;
                    node.turns = tentativeTurns;
                    node.h = calculateDistance(node, this.end);
                    node.updateGroupMembers = node.g + node.h;
                }
            }

            return undefined;
        }

        reconstructPath() {
            const K = [];
            for (let L = this.end; L !== this.start; L = L.previous) K.push(L);
            return K;
        }
    }

    class PathNode {
        constructor(K, L, M) {
            this.x = K;
            this.y = L;
            this.collision = M;
            this.g = 1e7;
            this.turns = 1e7;
            this.h = 1e7;
            this.updateGroupMembers = 1e7;
            this.previous = undefined;
            this.dir = -1;
        }
    }

    function idxOf(x, y) { return x + y * map.x; }
    function inBounds2(x, y) { return x >= 0 && y >= 0 && x < map.x && y < map.y; }
    function isWalkable(x, y) { return map.col.charAt(idxOf(x, y)) !== "1"; }

    function ensureRegionMap() {
        if (!window.map || !map.col) return null;

        const key = `${map.name}_${map.x}x${map.y}_${map.col.length}`;
        if (regionCache && regionKey === key) return regionCache;

        regionKey = key;

        const w = map.x, h = map.y;
        const region = new Int32Array(w * h);
        region.fill(-1);

        const q = new Int32Array(w * h);
        let regionId = 0;

        for (let start = 0; start < w * h; start++) {
            if (region[start] !== -1) continue;
            if (map.col.charAt(start) === "1") continue;

            let qs = 0, qe = 0;
            q[qe++] = start;
            region[start] = regionId;

            while (qs < qe) {
                const cur = q[qs++];
                const cx = cur % w;
                const cy = (cur / w) | 0;

                if (cy > 0) {
                    const ni = cur - w;
                    if (region[ni] === -1 && map.col.charAt(ni) !== "1") {
                        region[ni] = regionId;
                        q[qe++] = ni;
                    }
                }
                if (cy + 1 < h) {
                    const ni = cur + w;
                    if (region[ni] === -1 && map.col.charAt(ni) !== "1") {
                        region[ni] = regionId;
                        q[qe++] = ni;
                    }
                }
                if (cx > 0) {
                    const ni = cur - 1;
                    if (region[ni] === -1 && map.col.charAt(ni) !== "1") {
                        region[ni] = regionId;
                        q[qe++] = ni;
                    }
                }
                if (cx + 1 < w) {
                    const ni = cur + 1;
                    if (region[ni] === -1 && map.col.charAt(ni) !== "1") {
                        region[ni] = regionId;
                        q[qe++] = ni;
                    }
                }
            }

            regionId++;
        }

        regionCache = { key, region, w, h };
        return regionCache;
    }

    function getRegionAt(x, y) {
        const rc = ensureRegionMap();
        if (!rc) return -1;
        if (x < 0 || y < 0 || x >= rc.w || y >= rc.h) return -1;
        return rc.region[x + y * rc.w];
    }

    function getHeroRegion() {
        if (!window.hero) return -1;
        return getRegionAt(hero.x, hero.y);
    }

    function getStandTilesForPoint(x, y) {
        return [
            { x, y },
            { x: x + 1, y },
            { x: x - 1, y },
            { x, y: y + 1 },
            { x, y: y - 1 },
        ];
    }

    function isMobInHeroRegion(m) {
        const heroRegion = getHeroRegion();
        if (heroRegion < 0 || !m) return false;

        const spots = getStandTilesForPoint(m.x, m.y);
        for (let i = 0; i < spots.length; i++) {
            const s = spots[i];
            if (!inBounds2(s.x, s.y)) continue;
            if (!isWalkable(s.x, s.y)) continue;
            if (getRegionAt(s.x, s.y) === heroRegion) return true;
        }

        return false;
    }

    function canReachMapPointFromHeroRegion(x, y) {
        const heroRegion = getHeroRegion();
        if (heroRegion < 0) return false;

        return inBounds2(x, y) && isWalkable(x, y) && getRegionAt(x, y) === heroRegion;
    }

    function isGatewayInHeroRegion(gw) {
        if (!gw) return false;
        const heroRegion = getHeroRegion();
        if (heroRegion < 0) return false;

        const sx = Number(gw.standX);
        const sy = Number(gw.standY);

        if (!Number.isFinite(sx) || !Number.isFinite(sy)) return false;
        if (!inBounds2(sx, sy)) return false;
        if (!isWalkable(sx, sy)) return false;

        if (Number.isFinite(Number(gw.region)) && Number(gw.region) === heroRegion) {
            return true;
        }

        return getRegionAt(sx, sy) === heroRegion;
    }

    function buildNpcBlockedSetForBfs() {
        const set = new Set();
        try {
            if (!window.g || !g.npc || !window.map) return set;
            for (let id in g.npc) {
                const n = g.npc[id];
                if (!n) continue;
                if (!shouldTreatNormalMobsAsCollision() && n.wt === 1) continue;
                if (n.type === 4) continue;
                if (n.wt === -2) continue;

                const x = n.x, y = n.y;
                if (x < 0 || y < 0 || x >= map.x || y >= map.y) continue;
                const idx = x + y * map.x;
                if (map.col.charAt(idx) === "1") continue;
                set.add(idx);
            }
        } catch { }
        return set;
    }

    function bfsKey(sx, sy) {
        const npcCount = (window.g && g.npc) ? Object.keys(g.npc).length : 0;
        return `${map.name}_${map.x}x${map.y}_S${sx},${sy}_N${npcCount}`;
    }

    function getBfsDistFromHero() {
        const tNow = nowMs();
        if (!window.map || !window.hero) return null;

        const sx = hero.x, sy = hero.y;
        const key = bfsKey(sx, sy);
        if (bfsCache && bfsCache.key === key && (tNow - lastBfsAt) < BFS_CACHE_MS) return bfsCache;

        lastBfsAt = tNow;

        const w = map.x, h = map.y;
        const npcBlocked = buildNpcBlockedSetForBfs();

        const dist = new Int32Array(w * h);
        dist.fill(-1);

        const start = sx + sy * w;
        dist[start] = 0;

        const q = new Int32Array(w * h);
        let qs = 0, qe = 0;
        q[qe++] = start;

        while (qs < qe) {
            const cur = q[qs++];
            const cx = cur % w;
            const cy = (cur / w) | 0;
            const cd = dist[cur];

            if (cy > 0) {
                const ni = cur - w;
                if (dist[ni] === -1 && map.col.charAt(ni) !== "1" && !npcBlocked.has(ni)) {
                    dist[ni] = cd + 1; q[qe++] = ni;
                }
            }
            if (cy + 1 < h) {
                const ni = cur + w;
                if (dist[ni] === -1 && map.col.charAt(ni) !== "1" && !npcBlocked.has(ni)) {
                    dist[ni] = cd + 1; q[qe++] = ni;
                }
            }
            if (cx > 0) {
                const ni = cur - 1;
                if (dist[ni] === -1 && map.col.charAt(ni) !== "1" && !npcBlocked.has(ni)) {
                    dist[ni] = cd + 1; q[qe++] = ni;
                }
            }
            if (cx + 1 < w) {
                const ni = cur + 1;
                if (dist[ni] === -1 && map.col.charAt(ni) !== "1" && !npcBlocked.has(ni)) {
                    dist[ni] = cd + 1; q[qe++] = ni;
                }
            }
        }

        bfsCache = { key, dist, w, h, sx, sy };
        return bfsCache;
    }

    function dirToTarget(tx, ty) {
        const dx = tx - hero.x;
        const dy = ty - hero.y;
        if (Math.abs(dx) >= Math.abs(dy)) return dx >= 0 ? 2 : 3;
        return dy >= 0 ? 0 : 1;
    }

    function turnBonusForTarget(tx, ty) {
        if (lastMoveDir === null) return 0;
        const d = dirToTarget(tx, ty);
        if (d === lastMoveDir) return 12;
        const isOpposite =
            (lastMoveDir === 0 && d === 1) ||
            (lastMoveDir === 1 && d === 0) ||
            (lastMoveDir === 2 && d === 3) ||
            (lastMoveDir === 3 && d === 2);
        return isOpposite ? -10 : 0;
    }

    function getBestStandDistForMob(m, bfs) {
        if (!m || !bfs) return UNREACHABLE_DIST;

        const heroRegion = getHeroRegion();
        let best = UNREACHABLE_DIST;

        const spots = getStandTilesForPoint(m.x, m.y);
        for (let i = 0; i < spots.length; i++) {
            const s = spots[i];
            if (!inBounds2(s.x, s.y)) continue;
            if (!isWalkable(s.x, s.y)) continue;
            if (getRegionAt(s.x, s.y) !== heroRegion) continue;

            const d = bfs.dist[s.x + s.y * bfs.w];
            if (d >= 0 && d < best) best = d;
        }

        return best;
    }

    function getCandidateDist(m, bfs) {
        if (!m) return UNREACHABLE_DIST;
        if (!bfs) return Math.abs(hero.x - m.x) + Math.abs(hero.y - m.y);
        return getBestStandDistForMob(m, bfs);
    }

    let lastPickAt = 0;
    this.findBestMobEfficient = function () {
        if (!hasGameReady()) return null;

        const range = getLevelRange();
        if (!range) return null;

        syncDatabase();
        ensureRegionMap();

        const tNow = nowMs();
        if (tNow - lastPickAt < PICK_THROTTLE_MS) return B || null;
        lastPickAt = tNow;

        const ids = Object.keys(persistentDB);
        if (!ids.length) return null;

        cleanupTransientState(tNow);

        let bfs = getBfsDistFromHero();

        let pre = [];
        for (let i = 0; i < ids.length; i++) {
            const id = ids[i];
            if (I[id] && I[id] > tNow) continue;

            const m = persistentDB[id];
            if (!m) continue;
            if (!isMobInHeroRegion(m)) continue;

            pre.push({ id, d: getCandidateDist(m, bfs) });
        }

        if (pre.length === 0 && ids.length > 0) {
            bfsCache = null;
            lastBfsAt = 0;
            bfs = getBfsDistFromHero();

            for (let i = 0; i < ids.length; i++) {
                const id = ids[i];
                if (I[id] && I[id] > tNow) continue;

                const m = persistentDB[id];
                if (!m) continue;
                if (!isMobInHeroRegion(m)) continue;

                pre.push({ id, d: getCandidateDist(m, bfs) });
            }
        }

        if (pre.length === 0 && ids.length > 0 && tNow - lastIgnoreRecoveryAt > IGNORE_RECOVERY_COOLDOWN_MS) {
            lastIgnoreRecoveryAt = tNow;

            I = {};
            bfsCache = null;
            lastBfsAt = 0;
            bfs = getBfsDistFromHero();

            pre = [];
            for (let i = 0; i < ids.length; i++) {
                const id = ids[i];
                const m = persistentDB[id];
                if (!m) continue;
                if (!isMobInHeroRegion(m)) continue;
                pre.push({ id, d: getCandidateDist(m, bfs) });
            }
        }

        if (!pre.length) return null;

        pre.sort((a, b) => a.d - b.d);
        if (pre.length > MAX_EVAL) pre.length = MAX_EVAL;

        let bestId = null;
        let bestScore = Infinity;

        for (let i = 0; i < pre.length; i++) {
            const id = pre[i].id;
            const m = persistentDB[id];
            if (!m) continue;

            if (pre[i].d >= UNREACHABLE_DIST) {
                unreachableCount[id] = (unreachableCount[id] || 0) + 1;
                addFailurePenalty(id, 1);

                if (unreachableCount[id] >= 2) {
                    I[id] = tNow + UNREACHABLE_IGNORE_MS;
                    unreachableCount[id] = 0;
                }
                continue;
            } else {
                unreachableCount[id] = 0;
                reduceFailurePenalty(id, 0.4);
            }

            const turnB = turnBonusForTarget(m.x, m.y);
            const currentTargetBias = (id === B ? CURRENT_TARGET_BIAS : 0);
            const softPenalty = failurePenalty[id] || 0;
            const score =
                pre[i].d * 10 -
                turnB -
                currentTargetBias +
                softPenalty;

            if (score < bestScore) {
                bestScore = score;
                bestId = id;
            }
        }

        return bestId;
    };

    this.findBestGw = function () {
        if (!resolvedGateway) return null;
        if (!Number.isFinite(Number(resolvedGateway.x)) || !Number.isFinite(Number(resolvedGateway.y))) return null;

        const gw = {
            x: Number(resolvedGateway.x),
            y: Number(resolvedGateway.y),
            standX: Number.isFinite(Number(resolvedGateway.standX)) ? Number(resolvedGateway.standX) : null,
            standY: Number.isFinite(Number(resolvedGateway.standY)) ? Number(resolvedGateway.standY) : null,
            targetMap: resolvedGateway.targetMap || "",
            region: resolvedGateway.region ?? null,
            accessible: !!resolvedGateway.accessible,
        };

        if (!gw.accessible) return null;
        if (!isGatewayInHeroRegion(gw)) return null;
        return gw;
    };

    function handleStuckWithTarget() {
        if (!RUNNING || !G || !window.g || g.battle || g.dead || !window.hero) return;
        if (!B) return;

        const now = nowMs();

        if (lastPosX === null) {
            lastPosX = hero.x;
            lastPosY = hero.y;
            stuckSince = now;
            return;
        }

        const moved = hero.x !== lastPosX || hero.y !== lastPosY;
        if (moved) {
            lastPosX = hero.x;
            lastPosY = hero.y;
            stuckSince = now;
            targetStuckTries = 0;
            return;
        }

        if (now - stuckSince < 10_000) return;
        if (now - lastUnstuckTry < 4_000) return;
        lastUnstuckTry = now;

        const M = g.npc[B] || persistentDB[B];
        if (!M) { B = undefined; return; }

        targetStuckTries++;

        A = false;
        window.road = [];
        storePath(M.x, M.y);
        addFailurePenalty(B, 1.5);
        updateStatus(`STUCK -> re-path (${targetStuckTries})`, "#FFAA00");

        if (targetStuckTries >= 3) {
            const droppedId = B;
            I[droppedId] = now + UNREACHABLE_IGNORE_MS;

            B = undefined;
            A = false;
            z = false;
            window.road = [];
            targetStuckTries = 0;
            stuckSince = now;
        }
    }

    // =========================================================
    // LAST MOVE DIR
    // =========================================================
    function updateLastMoveDir() {
        if (!window.hero) return;

        if (lastHeroX === null) {
            lastHeroX = hero.x;
            lastHeroY = hero.y;
            return;
        }

        const dx = hero.x - lastHeroX;
        const dy = hero.y - lastHeroY;

        if (dx === 1 && dy === 0) lastMoveDir = 2;
        else if (dx === -1 && dy === 0) lastMoveDir = 3;
        else if (dx === 0 && dy === 1) lastMoveDir = 0;
        else if (dx === 0 && dy === -1) lastMoveDir = 1;

        lastHeroX = hero.x;
        lastHeroY = hero.y;
    }

    // =========================================================
    // MAIN OVERRIDE
    // =========================================================
    function hookParseInputOnce() {
        if (parseInputHooked) return;
        if (typeof window.parseInput !== "function") return;

        v.PI = window.parseInput;
        window.parseInput = function () {
            const L = v.PI.apply(this, arguments);

            if (RUNNING && window.g && !g.battle && !g.dead && G) {
                syncDatabase();
                updateLastMoveDir();
                if (handleBlockedGatewayAlert()) {
                    return L;
                }
                if (handlePendingGatewayTimeout()) {
                    return L;
                }
                recordMovementPosition();
                handleStuckWithTarget();
                refreshResolvedTravelTarget(false);
                clearRoadIfHighLevelMobTooClose();

                if (handleServerWorkflow()) {
                    return L;
                }

                refreshBlessingStatusUI();

                if (handleGoToMapCoordinate()) {
                    return L;
                }

                if (handleHuntMode()) {
                    return L;
                }

                if (handleWatcherRouteMode()) {
                    return L;
                }

                if (!B && !F) {
                    const nextMob = (isGoToMapMode() || isHuntMode() || isWatcherRouteMode() || isExplorerMode()) ? null : v.findBestMobEfficient();

                    if (!nextMob) {
                        const levelZoneVisitName = getLevelZoneVisitName();
                        const levelClearKey = levelZoneVisitName && window.map?.name
                            ? `${levelZoneVisitName}|${map.name}`
                            : "";
                        if (
                            levelClearKey &&
                            resolvedInsideZone &&
                            lastLevelZoneClearKey !== levelClearKey &&
                            markCurrentZoneMapVisited()
                        ) {
                            lastLevelZoneClearKey = levelClearKey;
                            resetServerRouteState();
                            refreshResolvedTravelTarget(true);
                            updateStatus(`Map clear: ${map.name}`, "#78DCE8");
                            return L;
                        }

                        const gw = v.findBestGw();

                        if (gw) {
                            if (isMapSettling()) {
                                clearMovementState();
                                updateStatus("WAITING MAP", "#FFCC00");
                                return L;
                            }

                            const nextHop = resolvedGateway?.targetMap || resolvedNextMap || "Gateway";
                            const finalTarget = resolvedTarget || "-";
                            updateStatus(`Going: ${nextHop} -> ${finalTarget}`, "#00FFFF");

                            const { x: goX, y: goY } = getGatewayUseTarget(gw);
                            const gatewayKey = getResolvedGatewayKey();

                            if (watchGatewayStanding(gatewayKey)) {
                                return L;
                            }

                            if (Number.isFinite(goX) && Number.isFinite(goY) && hero.x === goX && hero.y === goY) {
                                const tNow = nowMs();

                                if (tNow - lastGatewayUseAt >= GATEWAY_USE_COOLDOWN_MS) {
                                    lastGatewayUseAt = tNow;
                                    pendingGatewayKey = gatewayKey;
                                    pendingGatewayUntil = tNow + PENDING_GATEWAY_MS;
                                    pendingGatewayFromMap = String(map.name || "");
                                    pendingGatewayRequiresKey = !!resolvedGateway?.requiresKey;
                                    window.road = [];

                                    if (typeof window._g === "function") {
                                        movementTestEvent("gateway-walk", `walk to ${resolvedGateway?.targetMap || ""}`, {
                                            targetX: goX,
                                            targetY: goY,
                                        });
                                        _g("walk");
                                    }
                                }
                            } else {
                                const sameGatewayPending =
                                    hasPendingGatewayLock() &&
                                    pendingGatewayKey === gatewayKey;

                                if (!sameGatewayPending) {
                                    const needsNewRoad =
                                        Number.isFinite(goX) &&
                                        Number.isFinite(goY) &&
                                        !hasStoredPathFor(goX, goY);

                                    if (needsNewRoad) {
                                        const foundGatewayPath = storeGatewayPathOrTryAlternative(
                                            gatewayKey,
                                            goX,
                                            goY,
                                            isExplorerMode() ? "Explorer gateway" : "Gateway",
                                        );
                                        if (isExplorerMode() && !foundGatewayPath && (!window.road || !window.road.length)) {
                                            blockGatewayForCurrentHero(gatewayKey, "Explorer gateway path blocked -> trying another route");
                                            return L;
                                        }
                                        if (!window.road || !window.road.length) return L;
                                    }
                                }
                            }
                        } else {
                            if (routeFetchPending || nextStepRefreshInFlight) {
                                updateStatus("WAITING ROUTE", "#FFCC00");
                                return L;
                            }

                            if (nowMs() < serverFetchBackoffUntil && lastServerFetchSuccessAt > 0) {
                                updateStatus("WAITING SERVER", "#FFCC00");
                                return L;
                            }

                            clearMovementState();
                            updateStatus("IDLE", "#FFCC00");
                        }
                    } else {
                        setTarget(nextMob);
                        window.road = [];
                        updateStatus(`Target: ${persistentDB[B]?.name || "mob"}`, "#FFA500");
                    }
                }

                if (B) {
                    if (isWatcherRouteMode() && isCurrentHeroPartyNonLeader()) {
                        B = undefined;
                        A = false;
                        z = false;
                        window.road = [];
                        if (handleWatcherPatrol()) return L;
                        updateStatus("Watcher group no leader: patrol", "#e7b54a");
                        return L;
                    }

                    const M = g.npc[B] || persistentDB[B];
                    if (!M) {
                        B = undefined;
                        return L;
                    }

                    const dist = calculateDistance(hero, M);
                    if (dist < 2 && !z) {
                        updateStatus("BATTLE", "#FF0000");
                        z = true;
                        attackTarget(B);
                        reduceFailurePenalty(B, 2);
                        setTimeout(() => { B = undefined; }, 1000);
                    } else if ((!A || !window.road || window.road.length === 0) && !z) {
                        if (nowMs() - lastPathBuildAt >= PATH_BUILD_COOLDOWN_MS) {
                            lastPathBuildAt = nowMs();

                            const path = findPathToLocation(M.x, M.y);
                            if (!Array.isArray(path) || !path.length) {
                                const droppedId = B;

                                addFailurePenalty(droppedId, 2);
                                unreachableCount[droppedId] = (unreachableCount[droppedId] || 0) + 1;

                                if (unreachableCount[droppedId] >= 2) {
                                    I[droppedId] = nowMs() + UNREACHABLE_IGNORE_MS;
                                    unreachableCount[droppedId] = 0;
                                }

                                B = undefined;
                                A = false;
                                window.road = [];
                            } else {
                                window.road = path;
                                A = true;
                                reduceFailurePenalty(B, 1);
                                unreachableCount[B] = 0;
                            }
                        }
                    }
                }
            }

            return L;
        };

        parseInputHooked = true;
    }

    // BOOT

    async function boot() {
        if (bootStarted) return;
        bootStarted = true;

        await pullStorageVault();
        RUNNING = false;

        addStatusEvent("ClanHub zone script initialized");
        seenMobMeta = loadSeenMeta();
        v.initHTML();

        function waitForGame() {
            if (!hasGameReady()) {
                setTimeout(waitForGame, 200);
                return;
            }

            try {
                registerLoadQueueOnce();
            } catch (e) {
                console.error("xD script: loadQueue hook failed", e);
            }

            try {
                hookParseInputOnce();
            } catch (e) {
                console.error("xD script: parseInput hook failed", e);
            }

            updateStatus(RUNNING ? "RUNNING" : "READY", RUNNING ? "#00ff00" : "#ffff00");
            startLocalSyncIntervals();
        }

        waitForGame();
    }
